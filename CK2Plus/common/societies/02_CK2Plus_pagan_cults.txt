cult_of_cybele = {
	primary_attribute = stewardship
	is_secret = yes
	society_ranks_gfx = generic_ranks
	opinion_to_other_members = +10
	opinion_per_rank_above = +5
	sound = hermetic_society_interface
	non_interference = {
		always = yes
	}
	active = {
		has_dlc = "Mystics"
	}
	
	can_join_society = {
		hidden_trigger = {
			NAND = {
				ai = yes
				is_society_rank_full = { 
					society = cult_of_cybele
					rank = 1
				}
			}
		}
		stewardship = 10
		age = 16
		religion = hellenic_pagan
		custom_tooltip = {
			text = did_not_betray_society
			NOT = { has_character_flag = betrayed_cult_of_cybele }
		}
		scaled_wealth = 0.25
		hidden_trigger = {
			NOT = { has_character_flag = society_join_block }
		}
	}
	
	show_society = {
		religion = hellenic_pagan
	}

	potential = {
		religion = hellenic_pagan
		age = 16
	}

}
