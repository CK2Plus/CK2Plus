mysteries_of_isis = {
	primary_attribute = learning
	is_secret = yes
	is_religious = yes
	society_ranks_gfx = generic_ranks3
	opinion_to_other_members = +10
	opinion_per_rank_above = +5
	sound = hermetic_society_interface
	non_interference = {
		always = yes
	}
	active = {
		#temp_disabled
		always = no
	}
	
	can_join_society = {
		hidden_trigger = {
			NAND = {
				ai = yes
				is_society_rank_full = { 
					society = mysteries_of_isis
					rank = 1
				}
			}
		}
		learning = 10
		age = 16
		OR = {
			religion_group = hellenic_pagan_group
			religion = egyptian_pagan
		}
		custom_tooltip = {
			text = did_not_betray_society
			NOT = { has_character_flag = betrayed_mysteries_of_isis }
		}
		scaled_wealth = 0.25
		hidden_trigger = {
			NOT = { has_character_flag = society_join_block }
		}
	}
	
	show_society = {
		OR = {
			religion_group = hellenic_pagan_group
			religion = egyptian_pagan
		}
		society_member_of = mysteries_of_isis
	}

	potential = {
		OR = {
			religion_group = hellenic_pagan_group
			religion = egyptian_pagan
		}
		age = 16
	}
	
	#Normally, we'd have society currency defined here. We may still have to, but I'd prefer not to use it. Let's use timers instead.
	
	#"Non-clergy" members
	society_rank = {
		level = 1
		limit = 50 
		startup_limit = 0
		modifier = {
			monthly_character_piety = 0.05
		}
		decisions = {
			#This is something that should only be done once
			#First step in initiation, can be repated if not successful.
			mysteries_of_isis_purification_rite
			#Needs purification rite
			#Gives better connection to Isis and increases piety
			#If this one fails, character should be given a choice between leaving society permanently or getting some bad modifiers
			mysteries_of_isis_initiation_isis
			#needs isis initiation
			#Gives better connection to Osiris and increases piety
			mysteries_of_isis_initiation_osiris
			#Joining ritual
			#needs both isis and osiris initiation
			#if successful character will get better education, or gain at least tier-2 learning education trait
			mysteries_of_isis_joining_rite
		}
	}
	
	#Actual "priesthood"
	society_rank = {
		level = 2
		limit = 20
		startup_limit = 0
		modifier = { 
			monthly_character_piety = 0.1
			learning = 1
		}
		decisions = {
			#this shouldn't be a copy of hermetic scrying, otherwise we need to limit society by dlc
			#should have supernatural and non-supernatural version
			pagan_perform_scrying # Enables you to perform a scrying and gain an omen
			#Meditation or something, probably should include other priests
			mysteries_of_isis_divine_contact 
		}
	}
	
	#Cult Leader
	society_rank = {
		level = 3
		limit = 1
		startup_limit = 0
		modifier = {
			is_visible = {
				OR = {
					society_member_of = mysteries_of_isis
					FROM = { is_society_discovered = yes }
				}
			}
			monthly_character_piety = 0.5
			learning = 1
		}
		decisions = {
			#no special decision, but should have a blot like event every N years with all the priesthood
		}
	}
	
	#This means cult won't be created on game start.
	startup_populate = {
	    trigger = {
	    	always = no
		}
	}
}
