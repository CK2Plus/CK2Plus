# Child Realizes They Are Homosexual

character_event = {
	id = Plus.4050
	desc = "EVTDESC_Plus_4050"
	picture = "GFX_evt_child_play"
	
	min_age = 10
	max_age = 16

	trigger = {
		NOT = { has_global_flag = always_visible_traits } # NYI custom start option
		trait = homosexual
	}
	
	mean_time_to_happen = {
		days = 1 # for testing, probaby should be about 150 days
	}
	
	immediate = {
		hidden_tooltip = {
			remove_trait = homosexual
			add_trait = homosexual_visible
		}
	}
		
	option = {
		trigger = {
			has_guardian = yes
		}
		name = "EVTOPTA_PLUS_4050"
		custom_tooltip = { text = "PLUS_4050_hover_ctt" }
		hidden_tooltip = {
			guardian = {
				character_event = {
					id = Plus.4051
					days = 3
				}
			}
		}
	}
	
	option = {
		trigger = {
			has_guardian = no
		}
		name = "EVTOPTB_PLUS_4050"
		custom_tooltip = { text = "PLUS_4050_hover_ctt" }
	}
}
