character_event = { # The reaction to the menu as a hidden bounce event
					# Normally this would bounce to the emperor, then back to ROOT to establish
					# nice and tidy scopes for FROM and ROOT as well as establish the correct
					# character portraits on the event, however I will do that later
	id = Plus.4604
	
	hide_window = yes
	is_triggered_only = yes
	
	immediate = {
		random_list = {
			30 = { # Good Outcome
				modifier = {
					factor = 1.1
					OR = { # These are fairly standard fares
						has_character_flag = dt_feast_pig
						has_character_flag = dt_feast_poultry
					}
				}
				modifier = {
					factor = 1.2
					has_character_flag = dt_feast_salmon
				}
				modifier = {
					factor = 1.3
					has_character_flag = dt_feast_cattle
				}
				set_character_flag = dt_feast_good
			}
			40 = { # Neutral Outcome
				set_character_flag = dt_feast_neutral
			}
			30 = { # Bad Outcome
				modifier = {
					factor = 1.1
					OR = { # Dairy products are probably not ideal for East Asians
						has_character_flag = dt_feast_roasted_milk
						has_character_flag = dt_feast_cheese_eggs
					}
				}
				modifier = {
					factor = 1.3
					has_character_flag = dt_feast_horse
				}
				modifier = {
					factor = 1.2
					has_character_flag = dt_feast_balls
				}
				set_character_flag = dt_feast_bad
			}
		}
		narrative_event = { id = Plus.4605 }	# Should be 30 days gap here, but there is none 
												# for the purposes of testing
	}
}
