			activate_title = { title = d_finnish_pagan_papal status = yes }
			any_independent_ruler = { #indepentends, with a one day delay
				narrative_event = { id = Plus.4414 days = 1 }
			}
			#convert decision taker first!
			religion = finnish_pagan_papal
			set_global_flag = finnic_reformation
			custom_tooltip = {
				text = vassal #I don't think this is a correct localisation key
				hidden_tooltip = {
					job_spiritual = {
						d_finnish_pagan_papal = {
							grant_title = PREV
						}
						set_defacto_liege = ROOT
						pf_liege_change_effect = yes
						add_friend = ROOT
						religion = finnish_pagan_papal
					}
					activate_title = { title = d_finnish_pagan_reformed status = yes }
					activate_title = { title = d_finnish_pagan_autocephaly status = yes }
					set_parent_religion = { religion = finnish_pagan_papal parent = finnish_pagan }
					set_parent_religion = { religion = finnish_pagan_headless parent = finnish_pagan }
					set_parent_religion = { religion = finnish_pagan_autocephaly parent = finnish_pagan }

						limit = {
							religion = finnish_pagan
							is_landed = no #if they're landed they should get an event
						}
						religion = finnish_pagan_papal
					}
					any_courtier = {
						limit = {
							religion = finnish_pagan
							is_landed = no #exclude vassals from here.
						}
						religion = finnish_pagan_papal
					}
					any_vassal = {
						#vassals get a choice.
						#don't check if they're finnic, just send them event.
						narrative_event = { id = Plus.4414 days = 1 }
					}
					capital_scope = {
						#convert capital always
						if = {
							limit = { religion = finnish_pagan }
							religion = finnish_pagan_papal
						}
					}
					any_realm_province = {
						if = {
							limit = { religion = finnish_pagan }
							#leave some wiggle room, don't convert everything automatically.
							random_list = {
								20 = { #don't covert
									#if at least 4 or 5 holy sites are controled decrease resistance
									modifier = {
										factor = 0.5
										num_of_holy_sites = 4
									}
									modifier = {
										factor = 0.5
										num_of_holy_sites = 5
									}
								}
								80 = { #convert
									
									if = {
										limit = { religion = finnish_pagan }
										religion = finnish_pagan_papal
									}
								}
							}
						}
					}
				}
			}
		}

			factor = 1
			modifier = {
				factor = 0

					value = players
